name: {volume}
services:
  apache:
    container_name: {volume}
    build:
      context: apache
      dockerfile: ./{volume}-Dockerfile
    extra_hosts:
      - "{volume}.localhost:127.0.0.1"
    hostname: {volume}.localhost
    volumes: 
      - {volume}:/var/www/html
      - {volume}_db:/var/run/mysqld
      - {volume}_db:/var/lib/mysql
    ports:
      - "{webport}:80"
    depends_on:
      - mysql
    networks:
      - {volume}_network
  mysql:
    container_name: {volume}_db
    image: mysql:8.1.0
    ports:
      - "{sqlport}:3306"
    environment:
      - MYSQL_ROOT_PASSWORD={DBrootPWD}
      - MYSQL_USER={DBuser}
      - MYSQL_PASSWORD={DBuserPWD}
      - MYSQL_DATABASE={volume}
    volumes: 
      - {volume}_db:/var/run/mysqld
      - {volume}_db:/var/lib/mysql
    networks:
      {volume}_network:
        aliases:
          - database

  {volume}:
    name: "{volume}"
    driver: local
    driver_opts: 
      type: none
